77 const Value& Device::get_entry(const std::string& entry_name, uint8_t array_index, ReadAccessMethod access_method) {
78 
79  const std::string name = Utils::escape(entry_name);
80 
81  if (m_dictionary.count(name) == 0) {
82  throw dictionary_error(dictionary_error::type::unknown_entry, name);
83  }
84 
85  Entry& entry = m_dictionary[name];
86 
87  if (array_index > 0 && !entry.is_array) {
88  throw dictionary_error(dictionary_error::type::no_array, name);
89  }
90 
91  if (access_method==ReadAccessMethod::sdo || (access_method==ReadAccessMethod::use_default && entry.read_access_method==ReadAccessMethod::sdo)) {
92 
93  DEBUG_LOG("[Device::get_entry] update_on_read.");
94 
95  uint8_t subindex = (entry.is_array) ? 0x1+array_index : entry.subindex;
96  entry.set_value(get_entry_via_sdo(entry.index, subindex, entry.type), array_index);
97 
98  }
99 
100  return entry.get_value(array_index);
101

123 void Device::set_entry(const std::string& entry_name, const Value& value, uint8_t array_index, WriteAccessMethod access_method) {
124 
125  const std::string name = Utils::escape(entry_name);
126 
127  if (m_dictionary.count(name) == 0) {
128  throw dictionary_error(dictionary_error::type::unknown_entry, name);
129  }
130 
131  Entry& entry = m_dictionary[name];
132 
133  if (array_index > 0 && !entry.is_array) {
134  throw dictionary_error(dictionary_error::type::no_array, name);
135  }
136 
137  if (value.type != entry.type) {
138  throw dictionary_error(dictionary_error::type::wrong_type, name,
139  "Entry type: "+Utils::type_to_string(entry.type)+", given type: "+Utils::type_to_string(value.type));
140  }
141 
142  entry.set_value(value, array_index);
143 
144  if (access_method==WriteAccessMethod::sdo || (access_method==WriteAccessMethod::use_default && entry.write_access_method==WriteAccessMethod::sdo)) {
145 
146  DEBUG_LOG("[Device::set_entry] update_on_write.");
147 
148  const uint8_t subindex = (entry.is_array) ? 0x1+array_index : entry.subindex;
149  set_entry_via_sdo(entry.index, subindex, value);
150 
151  }

155 void Device::add_receive_pdo_mapping(uint16_t cob_id, const std::string& entry_name, uint8_t offset, uint8_t array_index) {
156

187 void Device::add_transmit_pdo_mapping(uint16_t cob_id, const std::vector<Mapping>& mappings, TransmissionType transmission_type, std::chrono::milliseconds repeat_time) {
188 
189  // Contructor can throw dictionary_error. Letting user handle this.
190  m_transmit_pdo_mappings.emplace_back(m_core, m_dictionary, cob_id, transmission_type, repeat_time, mappings);
191  TransmitPDOMapping& pdo = m_transmit_pdo_mappings.back();
192 
193  if (transmission_type==TransmissionType::ON_CHANGE) {
194 
195  for (const Mapping& mapping : pdo.mappings) {
196 
197  const std::string entry_name = Utils::escape(mapping.entry_name);
198 
199  // entry exists because check_correctness() == true.
200  Entry& entry = m_dictionary.at(entry_name);
201 
202  entry.add_value_changed_callback([entry_name, &pdo](const Value& value){
203  DEBUG_LOG("[Callback] Value of "<<entry_name<<" changed to "<<value);
204  pdo.send();
205  });
206  }
207 
208  } else {
209 
210  // transmission_type==TransmissionType::PERIODIC
211 
212  if (repeat_time == std::chrono::milliseconds(0)) {
213  WARN("[Device::add_transmit_pdo_mapping] Repeat time is 0. This could overload the bus.");
214  }
215 
216  pdo.transmitter = std::move(std::shared_ptr<std::thread>(new std::thread([&pdo, repeat_time](){
217 
218  while (true) {
219  DEBUG_LOG("[Timer thread] Sending periodic PDO.");
220  pdo.send();
221  std::this_thread::sleep_for(repeat_time);
222  }
223 
224  })));
225 
226  }
227 
228 }

230 void Device::pdo_received_callback(const ReceivePDOMapping& mapping, std::vector<uint8_t> data) {
231 
232  const std::string entry_name = Utils::escape(mapping.entry_name);
233  Entry& entry = m_dictionary[entry_name];
234  const uint8_t array_index = mapping.array_index;
235  const uint8_t offset = mapping.offset;
236  const uint8_t type_size = Utils::get_type_size(entry.type);
237 
238  if (data.size() < offset+type_size) {
239  // We don't throw an exception here, because this could be a network error.
240  ERROR("[Device::pdo_received_callback] PDO has wrong size!");
241  DUMP(data.size());
242  DUMP(offset);
243  DUMP(type_size);
244  }
245 
246  DEBUG_LOG("Updating entry "<<entry.name<<" (in case it's an array, index="<<array_index<<")");
247  std::vector<uint8_t> bytes(data.begin()+offset, data.begin()+offset+type_size);
248  entry.set_value(Value(entry.type,bytes), array_index);
249 
250 }

257 bool Device::load_dictionary_from_library() {
258 
259  if (m_eds_library.ready()) {
260 
261  uint16_t profile = get_device_profile_number();
262  uint32_t vendor_id = get_entry("Identity object/Vendor-ID");
263  uint32_t product_code = get_entry("Identity object/Product Code");
264  uint32_t revision_number = get_entry("Identity object/Revision number");
265 
266  m_dictionary.clear();
267  bool success = m_eds_library.load_manufacturer_eds(vendor_id, product_code, revision_number);
268 
269  if (success) {
270  DEBUG_LOG("[load_dictionary_from_library] Successfully loaded device specific dictionary: "<<std::dec<<vendor_id<<"/"<<product_code<<"."<<revision_number);
271  } else {
272  success = m_eds_library.load_default_eds(profile);
273  if (success) {
274  DEBUG_LOG("[load_dictionary_from_library] Successfully loaded generic CiA profile dictionary: cia/"<<std::dec<<profile<<".eds");
275  } else {
276  success = m_eds_library.load_mandatory_entries();
277  if (success) {
278  DEBUG_LOG("[load_dictionary_from_library] Successfully loaded mandatory CiA 301 entries");
279  } else {
280  ERROR("[load_dictionary_from_library] Could not automatically load the dictionary. Please manage dictionary by yourself.");
281  return false;
282  }
283  }
284  }
285 
286  } else {
287  ERROR("[Device::load_dictionary_from_library] EDS library not ready. Please manage dictionary by yourself.");
288  return false;
289  }
290 
291  return true;
292 
293 }
