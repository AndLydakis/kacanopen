64 void TransmitPDOMapping::send() const {
65 
66  std::vector<uint8_t> data(8,0);
67  size_t max_byte=0;
68 
69  DEBUG_LOG("[TransmitPDOMapping::send] Sending transmit PDO with cob_id 0x"<<std::hex<<cob_id);
70 
71  for (const Mapping& mapping : mappings) {
72 
73  const std::string entry_name = Utils::escape(mapping.entry_name);
74  const Entry& entry = m_dictionary.at(entry_name);
75  const Value& value = entry.get_value(mapping.array_index);
76  const std::vector<uint8_t> bytes = value.get_bytes();
77  assert(mapping.offset+bytes.size() <= 8);
78 
79  DEBUG_LOG("[TransmitPDOMapping::send] Mapping:");
80  DEBUG_DUMP(mapping.offset);
81  DEBUG_DUMP(entry_name);
82  DEBUG_DUMP_HEX(value);
83 
84  uint8_t count = 0;
85  for (uint8_t i=mapping.offset; i<mapping.offset+bytes.size(); ++i) {
86  data[i] = bytes[count++];
87  }
88 
89  max_byte = std::max(max_byte, mapping.offset+bytes.size());
90 
91  }
92 
93  data.resize(max_byte+1);
94  m_core.pdo.send(cob_id, data);
95 
96 }

98 void TransmitPDOMapping::check_correctness() const {
99 
100  unsigned size = 0;
101  std::vector<bool> byte_mapped(8,false);
102 
103  for (const Mapping& mapping : mappings) {
104 
105  const std::string entry_name = Utils::escape(mapping.entry_name);
106 
107  if (m_dictionary.count(entry_name) == 0) {
108  throw dictionary_error(dictionary_error::type::unknown_entry, entry_name);
109  }
110 
111  const Entry& entry = m_dictionary.at(entry_name);
112  const uint8_t type_size = Utils::get_type_size(entry.type);
113 
114  if (mapping.offset+type_size > 8) {
115  throw dictionary_error(dictionary_error::type::mapping_size, entry_name,
116  "mapping.offset ("+std::to_string(mapping.offset)+") + type_size ("+std::to_string(type_size)+") > 8.");
117  }
118 
119  if (mapping.array_index > 0 && !entry.is_array) {
120  throw dictionary_error(dictionary_error::type::no_array, entry_name);
121  }
122 
123  for (uint8_t i=mapping.offset; i<mapping.offset+type_size; ++i) {
124  if (byte_mapped[i]) {
125  throw dictionary_error(dictionary_error::type::mapping_overlap, entry_name, "Byte index: "+std::to_string(i));
126  }
127  byte_mapped[i] = true;
128  }
129 
130  size += type_size;
131 
132  }
133 
134  if (size > 8) {
135  throw dictionary_error(dictionary_error::type::mapping_size, "",
136  "total size ("+std::to_string(size)+") > 8.");
137  }
138 
139 }
