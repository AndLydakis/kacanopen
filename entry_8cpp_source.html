71 void Entry::set_value(const Value& value, uint8_t array_index) {
72 
73  if (value.type != type) {
74  ERROR("[Entry::set_value] You passed a value of wrong type.");
75  return;
76  }
77 
78  if (array_index>0 && !is_array) {
79  ERROR("[Entry::set_value] This is no array but you specified an array_index.");
80  return;
81  }
82 
83  bool value_changed = false;
84 
85  {
86  std::lock_guard<std::recursive_mutex> lock(*m_read_write_mutex);
87 
88  if (m_value.size()<=array_index) {
89  m_value.resize(array_index+1);
90  value_changed = true;
91  }
92 
93  if (m_valid.size()<=array_index) {
94  m_valid.resize(array_index+1, false);
95  value_changed = true;
96  }
97 
98  if (!value_changed && (m_value[array_index].type != type || m_value[array_index] != value) ) {
99  value_changed = true;
100  }
101 
102  m_value[array_index] = value;
103  m_valid[array_index] = true;
104  }
105 
106  if (value_changed) {
107  std::lock_guard<std::mutex> lock(*m_value_changed_callbacks_mutex);
108  for (auto& callback : m_value_changed_callbacks) {
109  // TODO: currently callbacks are only internal and it's ok to call them synchonously.
110  //std::async(std::launch::async, callback, value);
111  callback(value);
112  }
113  }
114 
115 }

130 bool Entry::valid(uint8_t array_index) const {
131 
132  if (array_index>0 && !is_array) {
133  ERROR("[Entry::valid] This is no array but you specified an array_index.");
134  return false;
135  }
136 
137  std::lock_guard<std::recursive_mutex> lock(*m_read_write_mutex);
138 
139  if ( array_index>=m_value.size()
140  || array_index>=m_valid.size()
141  || !m_valid[array_index] ) {
142  return false;
143  }
144 
145  return true;
