40 void Bridge::add_publisher(std::shared_ptr<Publisher> publisher, double loop_rate) {
41  m_publishers.push_back(publisher);
42  publisher->advertise();
43  m_futures.push_front(
44  std::async(std::launch::async, [publisher,loop_rate,this](){
45  ros::Rate rate(loop_rate);
46  while(ros::ok()) {
47  publisher->publish();
48  rate.sleep();
49  }
50  })
51  );
52 }
