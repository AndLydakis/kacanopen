44 int main(int argc, char* argv[]) {
45 
46  kaco::Master master;
47  bool success = master.start(BUSNAME, BAUDRATE);
48 
49  if (!success) {
50  ERROR("Starting master failed.");
51  return EXIT_FAILURE;
52  }
53 
54  std::this_thread::sleep_for(std::chrono::seconds(1));
55 
56  if (master.num_devices()<1) {
57  ERROR("No devices found.");
58  return EXIT_FAILURE;
59  }
60 
61  // should be a 401 device
62  kaco::Device& device = master.get_device(0);
63  device.start();
64  device.load_dictionary_from_library();
65  uint16_t profile = device.get_device_profile_number();
66 
67  if (profile != 401) {
68  ERROR("This example is intended for use with a CiA 401 device. You plugged a device with profile number "<<std::dec<<profile);
69  return EXIT_FAILURE;
70  }
71 
72  device.print_dictionary();
73 
74  DUMP(device.get_entry("Manufacturer device name"));
75 
76  // map PDOs (optional)
77  device.add_receive_pdo_mapping(0x188, "Read input 8-bit/Digital Inputs 1-8", 0); // offset 0
78  device.add_receive_pdo_mapping(0x188, "Read input 8-bit/Digital Inputs 9-16", 1); // offset 1
79 
80  // set some output (optional)
81  device.set_entry("Write output 8-bit/Digital Outputs 1-8", (uint8_t) 0xFF);
82 
83  ros::init(argc, argv, "canopen_bridge");
84 
85  // Create bridge
86  kaco::Bridge bridge;
87 
88  // create a publisher for reading second 8-bit input and add it to the bridge
89  // communication via POD
90  // publishing rate = 10 Hz
91  auto iopub = std::make_shared<kaco::EntryPublisher>(device, "Read input 8-bit/Digital Inputs 9-16");
92  bridge.add_publisher(iopub,1);
93 
94  // create a subscriber for editing IO output and add it to the bridge
95  // communication via SOD
96  auto iosub = std::make_shared<kaco::EntrySubscriber>(device, "Write output 8-bit/Digital Outputs 1-8");
97  bridge.add_subscriber(iosub);
98 
99  // run ROS loop
100  bridge.run();
101 
102  master.stop();
103 
104 }
