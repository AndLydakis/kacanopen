63 void NMT::process_incoming_message(const Message& message) {
64 
65  DEBUG_LOG("NMT Error Control message from node "
66  <<(unsigned)message.get_node_id()<<".");
67 
68  uint8_t data = message.data[0];
69  bool toggle_bit = data>>7;
70  uint8_t state = data&0x3F;
71 
72  //DEBUG_DUMP(toggle_bit);
73 
74  switch (state) {
75 
76  case 0: {
77  DEBUG_LOG("New state is Initialising");
78 
79  for (const auto& callback : m_new_device_callbacks) {
80  DEBUG_LOG("Calling new device callback (async)");
81  std::async(std::launch::async, callback, message.get_node_id());
82  }
83 
84  break;
85  }
86 
87  case 1: {
88  DEBUG_LOG("New state is Disconnected");
89  break;
90  }
91 
92  case 2: {
93  DEBUG_LOG("New state is Connecting");
94  break;
95  }
96 
97  case 3: {
98  DEBUG_LOG("New state is Preparing");
99  break;
100  }
101 
102  case 4: {
103  DEBUG_LOG("New state is Stopped");
104  break;
105  }
106 
107  case 5: {
108  DEBUG_LOG("New state is Operational");
109  break;
110  }
111 
112  case 127: {
113  DEBUG_LOG("New state is Pre-operational");
114  break;
115  }
116 
117  default: {
118  DEBUG_LOG("New state is unknown: "<<(unsigned)state);
119  break;
120  }
121 
122  }
