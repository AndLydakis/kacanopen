92 struct can_isotp_fc_options {
93 
94  __u8 bs; /* blocksize provided in FC frame */
95  /* __u8 value : blocksize. 0 = off */
96 
97  __u8 stmin; /* separation time provided in FC frame */
98  /* __u8 value : */
99  /* 0x00 - 0x7F : 0 - 127 ms */
100  /* 0x80 - 0xF0 : reserved */
101  /* 0xF1 - 0xF9 : 100 us - 900 us */
102  /* 0xFA - 0xFF : reserved */
103 
104  __u8 wftmax; /* max. number of wait frame transmiss. */
105  /* __u8 value : 0 = omit FC N_PDU WT */
106 };

108 struct can_isotp_ll_options {
109 
110  __u8 mtu; /* generated & accepted CAN frame type */
111  /* __u8 value : */

113  /* CANFD_MTU (72) -> CAN FD frame */
114 
115  __u8 tx_dl; /* tx link layer data length in bytes */
116  /* (configured maximum payload length) */
117  /* __u8 value : 8,12,16,20,24,32,48,64 */

120  __u8 tx_flags; /* set into struct canfd_frame.flags */
121  /* at frame creation: e.g. CANFD_BRS */
