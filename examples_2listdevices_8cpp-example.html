#include <thread>
#include <chrono>
#include <cstdint>
#include "master.h"
#include "logger.h"
// This example lists all connected devices and prints their manufacturer device name.
int main() {
PRINT("List devices");
kaco::Master master;
bool success = master.start(BUSNAME, BAUDRATE);
if (!success) {
ERROR("Starting master failed.");
return EXIT_FAILURE;
}
std::this_thread::sleep_for(std::chrono::seconds(1));
const size_t num_devices = master.num_devices();
DUMP(num_devices);
for (size_t i=0; i<num_devices; ++i) {
kaco::Device& device = master.get_device(i);
PRINT("Found device: " << device.get_node_id());
PRINT("Starting");
device.start();
PRINT("Loading EDS from library...")
device.load_dictionary_from_library();
PRINT("Dictionary:");
device.read_complete_dictionary();
device.print_dictionary();
//DUMP(device.get_entry("Manufacturer device name"));
}
}
